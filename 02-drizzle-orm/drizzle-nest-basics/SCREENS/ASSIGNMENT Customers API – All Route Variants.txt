## рҹ“Ғ DTOs

```ts
class ParamsDto {
    customerId: string;
    orderId: string;
    itemId: string;
}

class SearchCustomersDto {
    type?: 'PREMIUM' | 'NORMAL' | 'VIP';
    age?: number;
    city?: string;
}
```

---

## рҹ“Ғ customers.controller.ts

```ts
import {
    Controller,
    Get,
    Post,
    Body,
    Query,
    Delete,
    Param,
    Patch,
} from '@nestjs/common';

@Controller('customers')
export class CustomersController {

    // вң… 1. Query Params
    @Get()
    findAll(
        @Query('type') type?: string,
        @Query('age') age?: number
    ) {
        return {
            endpoint: 'GET /customers',
            queryReceived: {
                type: type ?? null,
                age: age ?? null
            }
        };
    }

    // вң… 2. Static Route
    @Get('vip')
    getVipCustomers() {
        return [
            { id: 1, name: 'VIP A' },
            { id: 2, name: 'VIP B' }
        ];
    }

    // вң… 3. Dynamic Route
    @Get(':id')
    findOne(@Param('id') id: string) {
        return {
            endpoint: 'GET /customers/:id',
            id
        };
    }

    // вң… 4. Nested Route
    @Get(':customerId/orders/:orderId')
    getOrder(
        @Param('customerId') customerId: string,
        @Param('orderId') orderId: string
    ) {
        return {
            customerId,
            orderId
        };
    }

    // вң… 5. POST
    @Post()
    create(@Body() body: any) {
        return {
            message: 'Customer created',
            body
        };
    }

    // вң… 6. PATCH
    @Patch(':id')
    update(@Param('id') id: string, @Body() body: any) {
        return {
            message: 'Customer updated',
            id,
            body
        };
    }

    // вң… 7. DELETE
    @Delete(':id')
    delete(@Param('id') id: string) {
        return {
            message: 'Customer deleted',
            id
        };
    }

    // вң… 8. Collect ALL Query Params
    @Get('search')
    search(@Query() query: SearchCustomersDto) {
        return {
            message: 'All query params',
            query
        };
    }

    // вң… 9. Collect ALL Params (DTO)
    @Get(':customerId/orders/:orderId/items/:itemId')
    getAllParams(@Param() params: ParamsDto) {
        return {
            message: 'All params DTO',
            params
        };
    }

    // вң… 10. Combine Params + Query
    @Get(':customerId/orders/:orderId/items/:itemId/full')
    getFull(
        @Param() params: any,
        @Query() query: any
    ) {
        return {
            params,
            query
        };
    }
}
```
