# 🧱 1. users/entity/user.entity.ts (REAL ENTITY)

👉 This is the **database table definition** in TypeORM

```ts
import { Entity, PrimaryGeneratedColumn, Column } from 'typeorm';

@Entity('nestjs_users_new')
export class User {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  name: string;

  @Column({ unique: true })
  email: string;

  @Column()
  password: string;

  @Column({ nullable: true })
  address: string;

  @Column({ nullable: true })
  profession: string;

  @Column({ nullable: true })
  occupation: string; // ✅ new field
}
```

# ⚙️ 4. users/users.service.ts (Repository Usage)

👉 THIS is where TypeORM is very different from Drizzle

```ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';

import { User } from './entities/user.entity';
import { CreateUserDto } from './dto/create-user.dto';
import { UpdateUserDto } from './dto/update-user.dto';

@Injectable()
export class UsersService {
  constructor(
    @InjectRepository(User)
    private readonly userRepository: Repository<User>,
  ) {}

  create(createUserDto: CreateUserDto) {
    const user = this.userRepository.create(createUserDto);
    return this.userRepository.save(user);
  }

  findAll() {
    return this.userRepository.find();
  }

  findOne(id: number) {
    return this.userRepository.findOneBy({ id });
  }

  update(id: number, updateUserDto: UpdateUserDto) {
    return this.userRepository.update(id, updateUserDto);
  }

  remove(id: number) {
    return this.userRepository.delete(id);
  }
}
```

# 🌐 5. users/users.controller.ts

```ts
import { Controller, Get, Post, Body, Patch, Param, Delete } from '@nestjs/common';
import { UsersService } from './users.service';

import { CreateUserDto } from './dto/create-user.dto';
import { UpdateUserDto } from './dto/update-user.dto';

@Controller('users')
export class UsersController {
  constructor(private readonly usersService: UsersService) {}

  @Post()
  create(@Body() createUserDto: CreateUserDto) {
    return this.usersService.create(createUserDto);
  }

  @Get()
  findAll() {
    return this.usersService.findAll();
  }

  @Get(':id')
  findOne(@Param('id') id: string) {
    return this.usersService.findOne(+id);
  }

  @Patch(':id')
  update(@Param('id') id: string, @Body() dto: UpdateUserDto) {
    return this.usersService.update(+id, dto);
  }

  @Delete(':id')
  remove(@Param('id') id: string) {
    return this.usersService.remove(+id);
  }
}
```

# 🗄️ 7. app.module.ts (DB Connection)

```ts
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { UsersModule } from './users/users.module';

@Module({
  imports: [
    TypeOrmModule.forRoot({
      type: 'postgres',
      url: process.env.DATABASE_URL,
      autoLoadEntities: true,
      synchronize: true, // ⚠️ dev only
      ssl: {
        rejectUnauthorized: false,
      },
    }),
    UsersModule,
  ],
})
export class AppModule {}
```

# 🧠 FINAL COMPARISON (VERY IMPORTANT)

| Feature     | TypeORM          | Drizzle      |
| ----------- | ---------------- | ------------ |
| Entity      | ✅ Required       | ❌ Not used   |
| Repository  | ✅ Required       | ❌ Not needed |
| Query style | Abstracted       | SQL-like     |
| Migration   | CLI + decorators | schema diff  |
